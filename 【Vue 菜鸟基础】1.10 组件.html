简单实例

<my-elem></my-elem>
<div id="app">
    <my-elem></my-elem>
</div>

<script>
    Vue.component('my-elem', {
        template: '<div><h1>自定义标题</h1><p>自定义内容</p></div>'
    })
</script>

局部组件

<!--局部组件，不可用-->
<my-elem></my-elem>

<div id="app">
    <!--局部组件只能在 #app 内部使用-->
    <my-elem></my-elem>
</div>

<script>
    var subComponent = {
        template: '<div><h1>自定义标题</h1><p>自定义内容</p></div>'
    }

    var app = new Vue({
        el: '#app',
        data: {

        },
        components: {
            'my-elem': subComponent
        }
    });
</script>

组件添加属性

<div id="app">
    <my-elem title="我是标题" content="我是内容"></my-elem>
</div>

<script>
    Vue.component('my-elem', {
        props:['title', 'content'],
        template: '<div><h1>{{title}}</h1><p>{{content}}</p></div>'
    })

</script>

动态绑定组件属性

  • 与常规元素，绑定属性类似；
<div id="app">
    <my-elem v-bind:title="myTitle" v-bind:content="myContent"></my-elem>
</div>

<script>

    Vue.component('my-elem', {
        props:['title', 'content'],
        template: '<div><h1>{{title}}</h1><p>{{content}}</p></div>'
    })

    var app = new Vue({
        el: '#app',
        data: {
            'myTitle': '我也是标题',
            'myContent': '我也是内容'
        }
    });
</script>

组件属性验证

    Vue.component('my-elem', {

        props: {
            // 指定类型
            title: String,
            
            // 多种类型
            content: [String, Number],
            
            // 指定类型，必填
            auth: {
                type: String,
                required: true
            },
            
            // 指定类型，默认值
            count: {
                type: Number,
                default: 100
            },
            
            // 指定类型，默认 对象
            info: {
                type:  Object,
                default: {
                    name: '张三',
                    age: 18
                }
            },
            
            // 指定类型，添加校验函数
            result: {
                validator: function (value) {
                    return value >= 60;
                }
            }
        },

        template: '<div><h1>{{title}}</h1><p>{{content}}</p></div>'
    })

自定义事件

  • 下面举例，对 my-elem 组件自定义了事件 my-click；
<div id="app">
    <my-elem v-on:my-click="instanceClick" v-on:click.native="nativeClick"></my-elem>
    <p>{{instanceMessage}}</p>
</div>

<script>
    Vue.component('my-elem', {
        template: '<div v-on:click="templateClick"><h1>{{title}}</h1><p>{{count}}</p></div>',

        data: function () {
            return {count: 0, title: '标题'}
        },

        methods: {
            templateClick: function () {
                this.count +=1;
                this.title = '修改标题';
                this.$emit('my-click');  // 触发自定义事件 my-click，实际是使用了组件中的 click 传递的，组件中使用 click 会被内部的 my-click 屏蔽，（可使用特殊处理）
            }
        }
    })

    var app = new Vue({
        el: '#app',
        data: {
            instanceMessage: 1
        },
        methods: {
            instanceClick: function () {
                this.instanceMessage *=2;
            }
            nativeClick: function () {
                alert('123');
            }
        }
    });

</script>

不屏蔽原始事件，使用 .native 修饰。

<div id="app">
    <my-elem v-on:my-click="instanceClick" v-on:click.native="nativeClick"></my-elem>
    <p>{{instanceMessage}}</p>
</div>
