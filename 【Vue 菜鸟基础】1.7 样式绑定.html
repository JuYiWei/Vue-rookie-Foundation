1.7 样式绑定

  • 准备
  • 绑定属性
  • v-bind:class= 绑定类型；

准备

    <style>
        .testClass1 {
            background: red;
        }

        .testClass2 {
            color: white;
        }
    </style>

绑定属性

  • 直接绑定 class
<div id="app">
    <p v-bind:class="class1">测试样式绑定</p>
    <p v-bind:class="{hasClass1? class1: ''}">测试样式绑定</p>
    <p v-bind:class="{testClass1: hasClass1}">测试样式绑定</p>
</div>

    var app = new Vue({
        el: '#app',
        data: {
            class1: 'testClass1',
            hasClass1: true,
        }
    });
  • 绑定多个 class
<div id="app">
    <p v-bind:class="[class1, class2]">测试样式绑定</p>
</div>

<script>
    var app = new Vue({
        el: '#app',
        data: {
            class1: 'testClass1',
            class2: 'testClass2',
        }
    });
</script>
  • 绑定对象
<div id="app">
    <p v-bind:class="styleObject">测试样式绑定</p>
</div>

<script>
    var app = new Vue({
        el: '#app',
        data: {
            hasClass1: true,
            hasClass2: true
        },
        computed: {
            styleObject: function () {
                return {
                    testClass1: this.hasClass1,
                    testClass2: this.hasClass2
                }
            }
        }
    });
</script>
  • 内联表达
<div id="app">
    <p v-bind:style="{'color': 'white', 'background': 'red'}">测试样式绑定</p>
</div>
