1.5 计算属性

  • 区别 computed 和 methods
  • 设置 setter 和 getter
  • computed 存在相互依赖关系时会重新取值，默认是 getter，也可以设置 setter；
  • 不是 methods，效果类似，但是它是函数；

区别 computed 和 methods

<div id="app">
    <p>{{title}}</p>
    <p>{{ucTitle}}</p>
    <p>{{ucTitle2()}}</p>
</div>

<script>
    var app = new Vue({
        el: '#app',
        data: {
            title: '我是标题'
        },
        computed: {
            ucTitle: function () {
                return '震惊！' + this.title;
            }
        },
        methods: {
            ucTitle2: function () {
                return '震惊！' + this.title;
            }
        }
    })
</script>

设置 setter 和 getter

<div id="app">
    <p>{{name}}</p>
    <p>{{age}}</p>
    <p>{{info}}</p>
</div>

<script>
    var app = new Vue({
        el: '#app',
        data: {
            name: '',
            age: 18
        },
        computed: {
            info: {
                set: function (newValue) {
                    var infos = newValue.split('今年');
                    this.name = infos[0];
                    this.age = infos[1];
                },
                get: function () {
                    return this.name + '今年' + this.age;
                }
            }
        }
    });

    app.info = 'LiLei今年19';
</script>
